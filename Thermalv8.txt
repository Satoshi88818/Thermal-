class SparseBilinear(nn.Module):
    """
    Linear layer with a fixed binary sparsity mask applied to the weight matrix.
    Mask is created once at init (random Bernoulli with given sparsity) and
    registered as a non-trainable buffer so it moves with the model to GPU.
    In practice this acts as structured dropout / regularisation; real hardware
    speedup would require custom CUDA sparse kernels.
    """
    def __init__(self, in_features: int, out_features: int,
                 block_size: int, sparsity: float = 0.9):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features) * 0.01)
        mask = torch.rand_like(self.weight) > sparsity
        self.register_buffer('mask', mask.float())
        self.weight.data.mul_(self.mask)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return F.linear(x, self.weight * self.mask)

def cosine_beta_schedule(T: int) -> np.ndarray:
    """
    Cosine noise schedule from Nichol & Dhariwal (2021).
    Produces smoother variance decay than linear schedules, preserving
    signal at low timesteps and avoiding over-destruction near t=T.
    """
    steps = np.arange(T + 1)
    alphas_cumprod = np.cos((steps / T) * np.pi * 0.5) ** 2
    alphas_cumprod = alphas_cumprod / alphas_cumprod[0]
    betas = 1 - alphas_cumprod[1:] / alphas_cumprod[:-1]
    return np.clip(betas, 0.0001, 0.9999)


class ScoreMatchingLoss:
    """
    Denoising Score Matching (DSM) loss.
    Computes ||score_theta(x_t, t) + noise / sqrt(1 - alpha_bar_t)||^2
    via automatic differentiation through the energy function.
    Note: score = -grad_x E(x, t) in the EBM convention used here.
    """
    def __call__(self, energy_fn: nn.Module,
                 x_t: torch.Tensor,
                 t: torch.Tensor,
                 noise: torch.Tensor,
                 model: 'BlockSparseDTM') -> torch.Tensor:
        score = torch.autograd.grad(
            energy_fn(x_t, t).sum(), x_t, create_graph=True
        )[0]
        B = x_t.shape[0]
        score = score.view(B, -1)
        noise_flat = noise.view(B, -1)
        alpha_bar = model.alpha_bar[t].view(-1, 1)
        target = -noise_flat / torch.sqrt(1.0 - alpha_bar)
        return F.mse_loss(score, target)

# 2. CONDITIONING PROCESSOR (U-Net style)
# =============================================
class CondProcessor(nn.Module):
    """
    Lightweight U-Net that converts a conditioning RGB image (64x64) into a
    spatial feature map of shape [B, 128, 8, 8] — one 128-d vector per fine
    block position.  Skip connections preserve multi-scale texture detail.
    """
    def __init__(self):
        super().__init__()
        # Encoder
        self.conv1 = nn.Conv2d(3,   16,  3, stride=1, padding=1)  # 64→64
        self.conv2 = nn.Conv2d(16,  32,  3, stride=2, padding=1)  # 64→32
        self.conv3 = nn.Conv2d(32,  64,  3, stride=2, padding=1)  # 32→16
        self.conv4 = nn.Conv2d(64,  128, 3, stride=2, padding=1)  # 16→8
        # Decoder
        self.up1   = nn.ConvTranspose2d(128, 64, kernel_size=2, stride=2)  # 8→16
        self.up2   = nn.ConvTranspose2d(64,  32, kernel_size=2, stride=2)  # 16→32
        self.up3   = nn.ConvTranspose2d(32,  16, kernel_size=2, stride=2)  # 32→64
        self.feat_out = nn.Conv2d(16, 128, 3, stride=1, padding=1)         # 64→64, 128ch
        self.bn1   = nn.BatchNorm2d(16)
        self.bn2   = nn.BatchNorm2d(32)
        self.bn3   = nn.BatchNorm2d(64)
        self.bn4   = nn.BatchNorm2d(128)

    def forward(self, cond_image: torch.Tensor) -> torch.Tensor:
        # cond_image: [B, 3, 64, 64]
        d1 = F.relu(self.bn1(self.conv1(cond_image)))  # [B,  16, 64, 64]
        d2 = F.relu(self.bn2(self.conv2(d1)))          # [B,  32, 32, 32]
        d3 = F.relu(self.bn3(self.conv3(d2)))          # [B,  64, 16, 16]
        d4 = F.relu(self.bn4(self.conv4(d3)))          # [B, 128,  8,  8]
        u1 = F.relu(self.up1(d4) + d3)                # [B,  64, 16, 16]
        u2 = F.relu(self.up2(u1) + d2)                # [B,  32, 32, 32]
        u3 = F.relu(self.up3(u2) + d1)                # [B,  16, 64, 64]
        feat = self.feat_out(u3)                       # [B, 128, 64, 64]
        # Pool to one vector per 8x8 fine block → [B, 128, 8, 8]
        cond_feat = F.avg_pool2d(feat, kernel_size=8, stride=8)
        return cond_feat  # [B, 128, 8, 8]


# =============================================
# 3. DILATED BLOCK CHAIN
# =============================================
class DilatedBlockChain(nn.Module):
    """
    Builds a context hidden state for each of the N=64 fine blocks using
    a series of dilated mixing passes, achieving O(log N) depth with full
    parallelism within each pass.

    Intuition (WaveNet-style applied spatially):
        Pass d=1  : each block sees its immediate neighbours
        Pass d=2  : each block sees blocks 2 apart  (via d=1 state)
        Pass d=4  : each block sees blocks 4 apart  (via d=2 state)
        ...
        Pass d=32 : every block has an effective receptive field of 64

    The mixing at each dilation step is a learned gated projection:
        gate   = sigmoid(W_g * [h_i, h_{i-d}])
        update = tanh   (W_u * [h_i, h_{i-d}])
        h_i    = (1 - gate) * h_i + gate * update

    This is equivalent to a linear-time GRU unrolled in the dilation tree,
    and is fully parallel across all N blocks in each pass (no data dependency
    within a single dilation level).

    [NEW] Outputs thermal field τ per block (softplus projection + offset for positivity).
    Heat equation initialization could be added here (e.g., set gates to average neighbors).

    Input  blocks : [B, N, input_dim]  (N = NUM_FINE_BLOCKS = 64)
    Output context: [B, N, CHAIN_HIDDEN], τ: [B, N]
    """
    def __init__(self, input_dim: int, hidden_dim: int = CHAIN_HIDDEN,
                 dilations: List[int] = CHAIN_DILATIONS):
        super().__init__()
        self.hidden_dim = hidden_dim
        self.dilations  = dilations

        # Project input tokens to hidden_dim
        self.input_proj = nn.Linear(input_dim, hidden_dim)

        # One gated mixing layer per dilation level
        # Each takes [h_i || h_{i-d}] → gate and update
        self.gate_projs   = nn.ModuleList([
            nn.Linear(hidden_dim * 2, hidden_dim) for _ in dilations
        ])
        self.update_projs = nn.ModuleList([
            nn.Linear(hidden_dim * 2, hidden_dim) for _ in dilations
        ])
        self.layer_norms  = nn.ModuleList([
            nn.LayerNorm(hidden_dim) for _ in dilations
        ])

        # [NEW] Thermal field projection
        self.tau_proj = nn.Linear(hidden_dim, 1)

        # TODO: Optional heat equation init (e.g., set weights to approximate Laplacian averaging)

    def forward(self, blocks: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        blocks: [B, N, input_dim]
        returns context: [B, N, hidden_dim], tau: [B, N]
        """
        B, N, _ = blocks.shape
        # Project to hidden space: [B, N, H]
        h = F.relu(self.input_proj(blocks))

        for i, d in enumerate(self.dilations):
            # Gather dilated neighbour (zero-pad out-of-bounds positions)
            # h_shifted[n] = h[n - d] if n >= d, else zeros
            # This is a simple tensor slice + zero-pad — fully parallel.
            if d < N:
                h_shifted = torch.zeros_like(h)
                h_shifted[:, d:, :] = h[:, :-d, :]   # shift right by d
            else:
                # Dilation larger than sequence — no neighbour available
                h_shifted = torch.zeros_like(h)

            # Concatenate self with dilated neighbour: [B, N, 2H]
            h_cat = torch.cat([h, h_shifted], dim=-1)

            # Gated update — all N blocks in parallel
            gate   = torch.sigmoid(self.gate_projs[i](h_cat))    # [B, N, H]
            update = torch.tanh(self.update_projs[i](h_cat))     # [B, N, H]
            h      = self.layer_norms[i]((1.0 - gate) * h + gate * update)

        # [NEW] Compute thermal field τ > 0
        tau = F.softplus(self.tau_proj(h)).squeeze(-1) + 0.1  # [B, N]

        return h, tau  # context h, thermal τ

# 4. ENERGY FUNCTION
# =============================================
class BlockSparseEnergy(nn.Module):
    """
    Per-block energy function: a deep sparse MLP with skip connection.
    For fine blocks: cond_dim = 128 (CondProcessor) + CHAIN_HIDDEN (dilated chain).
    For coarse blocks: no conditioning (cond_dim = 0).

    Energy is defined as:
        E(x, t) = -(MLP_local(x, cond) + 0.3 * MLP_skip(x, cond)).sum()
                  - bias · x
                  - t_bias · x
    Lower energy = higher probability under the EBM convention used here.
    """
    def __init__(self, features: int, sparsity: float = 0.9,
                 is_fine: bool = False, chain_hidden: int = CHAIN_HIDDEN):
        super().__init__()
        hidden = 2048
        self.is_fine  = is_fine
        # Fine blocks get: spatial cond (128) + chain context (CHAIN_HIDDEN)
        cond_dim = (128 + chain_hidden) if is_fine else 0
        in_feat  = features + cond_dim

        self.local = nn.Sequential(
            SparseBilinear(in_feat, hidden, block_size=in_feat, sparsity=sparsity),
            nn.LayerNorm(hidden), nn.ReLU(),
            SparseBilinear(hidden, hidden, block_size=hidden, sparsity=sparsity),
            nn.LayerNorm(hidden), nn.ReLU(),
            SparseBilinear(hidden, hidden, block_size=hidden, sparsity=sparsity),
            nn.LayerNorm(hidden), nn.ReLU(),
            SparseBilinear(hidden, features, block_size=hidden, sparsity=sparsity),
        )
        self.skip = nn.Sequential(
            SparseBilinear(in_feat, hidden, block_size=in_feat, sparsity=0.99),
            nn.LayerNorm(hidden), nn.ReLU(),
            SparseBilinear(hidden, features, block_size=hidden, sparsity=0.99),
        )
        self.bias = nn.Parameter(torch.zeros(features))

    def forward(self, x_block: torch.Tensor,
                t_bias: torch.Tensor,
                cond: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        x_block : [B', features]        B' = B * num_blocks (blocks batched together)
        t_bias  : [B', features]
        cond    : [B', cond_dim] or None
        returns : [B'] scalar energy per block
        """
        if cond is not None:
            input_x = torch.cat([x_block, cond], dim=-1)
        else:
            input_x = x_block
        h1     = self.local(input_x)
        h2     = self.skip(input_x)
        energy = (
            -(h1 + 0.3 * h2).sum(dim=-1)
            - (self.bias * x_block).sum(dim=-1)
            - (t_bias * x_block).sum(dim=-1)
        )
        return energy  # [B']

# 5. DTM MODEL
# =============================================
class BlockSparseDTM(nn.Module):
    """
    Block-Sparse Discrete Thermal Model with Dilated Block Chaining.

    Architecture overview:
        ┌─────────────────────────────────────────────┐
        │  Input x0 [B,3,64,64]                       │
        │      │                                       │
        │  CondProcessor → cond_feat [B,128,8,8]      │
        │      │                                       │
        │  Extract fine blocks → [B,64,192]            │
        │      │                                       │
        │  DilatedBlockChain → chain_ctx [B,64,256]   │  ← NEW
        │      │                                       │
        │  BlockSparseEnergy (fine, per block)         │
        │    input = [block(192) | cond(128) | ctx(256)]│
        │      │                                       │
        │  BlockSparseEnergy (coarse, per block)       │
        │    input = [block(768)]                      │
        │      │                                       │
        │  Total energy = sum coarse + sum fine        │
        └─────────────────────────────────────────────┘

    Training: DSM loss on add_noise → compute_energy.
    Sampling: DDPM-style reverse diffusion with predicted x0 from score.
    """
    def __init__(self):
        super().__init__()
        self.T         = T_STEPS
        self.betas     = BETAS
        self.alphas    = ALPHAS
        self.alpha_bar = ALPHA_BAR

        # Time embedding
        self.time_emb        = nn.Embedding(T_STEPS, 32)
        self.time_proj_coarse = nn.Linear(32, PBITS_PER_COARSE_BLOCK)
        self.time_proj_fine   = nn.Linear(32, PBITS_PER_FINE_BLOCK)

        # Energy functions
        self.energy_fn_coarse = BlockSparseEnergy(
            PBITS_PER_COARSE_BLOCK, is_fine=False)
        self.energy_fn_fine   = BlockSparseEnergy(
            PBITS_PER_FINE_BLOCK,   is_fine=True, chain_hidden=CHAIN_HIDDEN)

        # Conditioning processor
        self.cond_processor   = CondProcessor()

        # [NEW] Dilated block chain
        # Input dim: PBITS_PER_FINE_BLOCK (192) raw block pixels as initial token
        self.dilated_chain    = DilatedBlockChain(
            input_dim=PBITS_PER_FINE_BLOCK,
            hidden_dim=CHAIN_HIDDEN,
            dilations=CHAIN_DILATIONS,
        )

        # Learnable temperature scale (kept for compatibility; applied in energy)
        self.temp_scale = nn.Parameter(torch.tensor(1.0))

        # [NEW] Thermal loss weights
        self.lambda_fe = LAMBDA_FE
        self.lambda_smooth = LAMBDA_SMOOTH

        self.coarse_grid = (COARSE_BLOCKS_PER_SIDE, COARSE_BLOCKS_PER_SIDE)
        self.fine_grid   = (FINE_BLOCKS_PER_SIDE,   FINE_BLOCKS_PER_SIDE)

    # ------------------------------------------------------------------
    # Helpers: block extraction / reconstruction
    # ------------------------------------------------------------------
    @staticmethod
    def _extract_blocks(x: torch.Tensor,
                        block_size: int,
                        blocks_per_side: int) -> torch.Tensor:
        """
        Rearrange image pixels into non-overlapping blocks.
        x     : [B, C, H, W]
        return: [B, num_blocks, block_size*block_size*C]  (raster order)
        """
        B, C, H, W = x.shape
        bps = blocks_per_side
        bs  = block_size
        # Reshape to [B, C, bps, bs, bps, bs]
        x = x.view(B, C, bps, bs, bps, bs)
        # Permute to [B, bps, bps, bs, bs, C] then flatten block dims
        x = x.permute(0, 2, 4, 3, 5, 1).contiguous()  # [B,bps,bps,bs,bs,C]
        x = x.view(B, bps * bps, bs * bs * C)
        return x  # [B, num_blocks, features]

    @staticmethod
    def _reconstruct_image(blocks: torch.Tensor,
                           block_size: int,
                           blocks_per_side: int,
                           img_size: int) -> torch.Tensor:
        """
        Inverse of _extract_blocks.
        blocks : [B, num_blocks, features]
        return : [B, C, img_size, img_size]
        """
        B   = blocks.shape[0]
        bps = blocks_per_side
        bs  = block_size
        C   = CHANNELS
        # [B, bps, bps, bs, bs, C]
        x = blocks.view(B, bps, bps, bs, bs, C)
        x = x.permute(0, 5, 1, 3, 2, 4).contiguous()  # [B,C,bps,bs,bps,bs]
        x = x.view(B, C, img_size, img_size)
        return x

    # ------------------------------------------------------------------
    # [NEW] Get thermal field τ from an image
    # ------------------------------------------------------------------
    def get_tau(self, x: torch.Tensor) -> torch.Tensor:
        """
        Extract fine blocks and run dilated chain to get τ [B, 64]
        """
        fine_blocks = self._extract_blocks(
            x, FINE_BLOCK_SIZE, FINE_BLOCKS_PER_SIDE)  # [B,64,192]
        _, tau = self.dilated_chain(fine_blocks)       # [B,64]
        return tau

    # ------------------------------------------------------------------
    # [NEW] Compute smoothness loss for τ (conductivity regularization)
    # ------------------------------------------------------------------
    def compute_smooth_tau(self, tau: torch.Tensor) -> torch.Tensor:
        """
        tau: [B, 64]
        returns scalar smoothness loss (mean squared diff of neighbors in 2D grid)
        """
        B = tau.shape[0]
        tau_grid = tau.view(B, FINE_BLOCKS_PER_SIDE, FINE_BLOCKS_PER_SIDE)  # [B,8,8]
        h_diff = (tau_grid[:, :, 1:] - tau_grid[:, :, :-1]).pow(2).mean()
        v_diff = (tau_grid[:, 1:, :] - tau_grid[:, :-1, :]).pow(2).mean()
        return (h_diff + v_diff) / 2

    # ------------------------------------------------------------------
    # Forward noise addition (global schedule; adaptive left as optional)
    # ------------------------------------------------------------------
    def add_noise(self, x0: torch.Tensor,
                  t: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Standard DDPM forward process:
            x_t = sqrt(alpha_bar_t) * x0 + sqrt(1 - alpha_bar_t) * eps
        Returns (x_t, eps).
        TODO: Make adaptive with τ (e.g., alpha_bar_eff = alpha_bar ** tau_map)
        """
        noise           = torch.randn_like(x0)
        sqrt_ab         = self.alpha_bar[t].view(-1, 1, 1, 1).sqrt()
        sqrt_one_minus  = (1.0 - self.alpha_bar[t]).view(-1, 1, 1, 1).sqrt()
        x_t             = sqrt_ab * x0 + sqrt_one_minus * noise
        return x_t, noise

    # ------------------------------------------------------------------
    # Energy computation (full image)
    # ------------------------------------------------------------------
    def compute_energy(self, x_t: torch.Tensor,
                       t: torch.Tensor,
                       cond_image: Optional[torch.Tensor] = None
                       ) -> torch.Tensor:
        """
        Compute the total scalar energy for a batch of noisy images.
        [NEW] Scale per-block energy by 1/τ (higher temp flattens distribution)

        x_t        : [B, 3, 64, 64]  noisy image at timestep t
        t          : [B]              integer timestep indices
        cond_image : [B, 3, 64, 64]  optional conditioning image
                     (if None, uses x_t itself as unconditional reference)
        returns    : [B] total energy per sample
        """
        B = x_t.shape[0]

        # Time bias vectors
        t_emb         = self.time_emb(t)                              # [B, 32]
        t_bias_coarse = self.time_proj_coarse(t_emb)                  # [B, 768]
        t_bias_fine   = self.time_proj_fine(t_emb)                    # [B, 192]

        # ── Coarse energy ──────────────────────────────────────────────
        coarse_blocks = self._extract_blocks(
            x_t, COARSE_BLOCK_SIZE, COARSE_BLOCKS_PER_SIDE)          # [B,16,768]
        # Expand t_bias to match all coarse blocks
        t_bias_c_exp  = t_bias_coarse.unsqueeze(1).expand(
            -1, NUM_COARSE_BLOCKS, -1).reshape(-1, PBITS_PER_COARSE_BLOCK)
        coarse_flat   = coarse_blocks.reshape(-1, PBITS_PER_COARSE_BLOCK)
        e_coarse_raw  = self.energy_fn_coarse(
            coarse_flat, t_bias_c_exp)                                # [B*16]

        # ── Fine energy with dilated chain conditioning ─────────────────
        fine_blocks   = self._extract_blocks(
            x_t, FINE_BLOCK_SIZE, FINE_BLOCKS_PER_SIDE)              # [B,64,192]

        # [NEW] Build chain context + thermal τ
        chain_ctx, tau = self.dilated_chain(fine_blocks)              # [B,64,256], [B,64]

        # Spatial conditioning from CondProcessor
        cond_src      = cond_image if cond_image is not None else x_t
        cond_feat     = self.cond_processor(cond_src)                 # [B,128,8,8]
        # Flatten spatial dim: [B,128,8,8] → [B,64,128]  (8x8=64 positions)
        cond_flat_spatial = cond_feat.permute(0, 2, 3, 1).reshape(
            B, NUM_FINE_BLOCKS, 128)                                  # [B,64,128]

        # Concatenate spatial cond + chain context → [B,64,384]
        full_cond     = torch.cat([cond_flat_spatial, chain_ctx], dim=-1)  # [B,64,384]

        # Flatten block and batch dims for energy function
        t_bias_f_exp  = t_bias_fine.unsqueeze(1).expand(
            -1, NUM_FINE_BLOCKS, -1).reshape(-1, PBITS_PER_FINE_BLOCK)
        fine_flat     = fine_blocks.reshape(-1, PBITS_PER_FINE_BLOCK)
        full_cond_flat = full_cond.reshape(-1, 128 + CHAIN_HIDDEN)    # [B*64,384]

        e_fine_raw    = self.energy_fn_fine(
            fine_flat, t_bias_f_exp, cond=full_cond_flat)             # [B*64]

        # [NEW] Downsample τ to coarse grid for coarse scaling
        tau_grid = tau.view(B, FINE_BLOCKS_PER_SIDE, FINE_BLOCKS_PER_SIDE).unsqueeze(1)  # [B,1,8,8]
        tau_coarse_grid = F.avg_pool2d(tau_grid, kernel_size=2, stride=2)  # [B,1,4,4]
        tau_coarse = tau_coarse_grid.squeeze(1).view(B, NUM_COARSE_BLOCKS)  # [B,16]

        # [NEW] Scale energies by 1/τ per block
        tau_fine_flat = tau.view(-1)  # [B*64]
        e_fine_scaled = e_fine_raw / tau_fine_flat
        e_fine = e_fine_scaled.view(B, NUM_FINE_BLOCKS).sum(dim=1)    # [B]

        tau_coarse_flat = tau_coarse.view(-1)  # [B*16]
        e_coarse_scaled = e_coarse_raw / tau_coarse_flat
        e_coarse = e_coarse_scaled.view(B, NUM_COARSE_BLOCKS).sum(dim=1)  # [B]

        return (e_coarse + e_fine) * self.temp_scale                  # [B]

    # ------------------------------------------------------------------
    # Score (gradient of energy w.r.t. x)
    # ------------------------------------------------------------------
    def compute_score(self, x_t: torch.Tensor,
                      t: torch.Tensor,
                      cond_image: Optional[torch.Tensor] = None
                      ) -> torch.Tensor:
        """
        score = -grad_x E(x_t, t)   shape: [B, 3, 64, 64]
        Requires x_t.requires_grad = True.
        """
        x_t.requires_grad_(True)
        energy = self.compute_energy(x_t, t, cond_image)
        score  = -torch.autograd.grad(
            energy.sum(), x_t, create_graph=self.training
        )[0]
        return score

    # ------------------------------------------------------------------
    # DSM training loss
    # ------------------------------------------------------------------
    def dsm_loss(self, x0: torch.Tensor,
                 cond_image: Optional[torch.Tensor] = None) -> torch.Tensor:
        """
        Randomly sample t ~ Uniform[0, T-1], add noise, then compute DSM loss:
            L = E_t E_{eps} || score_theta(x_t, t) + eps/sqrt(1-alpha_bar_t) ||^2
        [NEW] + λ_fe * (E - τ * entropy_term) + λ_smooth * smooth(τ)
        Where entropy_term ≈ -S via score norm proxy.
        """
        B   = x0.shape[0]
        t   = torch.randint(0, self.T, (B,), device=x0.device)
        x_t, noise = self.add_noise(x0, t)
        x_t = x_t.detach().requires_grad_(True)

        score      = self.compute_score(x_t, t, cond_image)           # [B,3,64,64]
        alpha_bar  = self.alpha_bar[t].view(-1, 1, 1, 1)
        target     = -noise / (1.0 - alpha_bar).sqrt()
        dsm_term   = F.mse_loss(score, target)

        # [NEW] Free energy components
        energy     = self.compute_energy(x_t, t, cond_image)          # [B]
        tau        = self.get_tau(cond_image if cond_image is not None else x0)  # [B,64] from clean
        entropy_term = score.pow(2).mean(dim=[1,2,3])                 # [B], proxy for -S
        free_energy_loss = energy.mean() - tau.mean() * entropy_term.mean()

        # [NEW] Conductivity regularization (smooth τ)
        smooth_loss = self.compute_smooth_tau(tau)

        # Total
        total_loss = dsm_term + self.lambda_fe * free_energy_loss + self.lambda_smooth * smooth_loss
        return total_loss

    # ------------------------------------------------------------------
    # DDPM-style reverse diffusion sampling
    # ------------------------------------------------------------------
    @torch.no_grad()
    def sample(self, n_samples: int = 4,
               cond_image: Optional[torch.Tensor] = None,
               verbose: bool = False) -> torch.Tensor:
        """
        Ancestral sampling from x_T ~ N(0,I) down to x_0.
        Uses DDPM reverse step:
            x_{t-1} = (1/sqrt(alpha_t)) * (x_t + beta_t * score(x_t, t))
                      + sqrt(beta_t) * z,    z ~ N(0,I) if t>0 else 0

        Returns x_0 : [n_samples, 3, 64, 64] in range [-1, 1].
        """
        x = torch.randn(n_samples, CHANNELS, IMG_SIZE, IMG_SIZE, device=DEVICE)

        for step in reversed(range(self.T)):
            t_tensor = torch.full((n_samples,), step,
                                  dtype=torch.long, device=DEVICE)
            x_in   = x.detach().requires_grad_(True)
            score  = self.compute_score(x_in, t_tensor, cond_image)
            score  = score.detach()

            beta_t  = self.betas[step]
            alpha_t = self.alphas[step]

            # Predict x_0 from score
            ab      = self.alpha_bar[step]
            x0_pred = (x + (1.0 - ab).sqrt() * score) / ab.sqrt()
            x0_pred = x0_pred.clamp(-1.0, 1.0)

            # Reverse step mean
            if step > 0:
                ab_prev = self.alpha_bar[step - 1]
            else:
                ab_prev = torch.ones(1, device=DEVICE)

            mean = (ab_prev.sqrt() * beta_t * x0_pred +
                    alpha_t.sqrt() * (1.0 - ab_prev) * x) / (1.0 - ab)

            if step > 0:
                noise = torch.randn_like(x)
                var   = (1.0 - ab_prev) / (1.0 - ab) * beta_t
                x     = mean + var.sqrt() * noise
            else:
                x     = mean

            if verbose and step % 20 == 0:
                print(f"  Sampling step {step:3d} / {self.T} done")

        return x.clamp(-1.0, 1.0)

def get_cifar10_loader(batch_size: int = BATCH_SIZE) -> DataLoader:
    """
    CIFAR-10 DataLoader, resized to IMG_SIZE=64 and normalised to [-1, 1].
    Downloads automatically on first run.
    """
    transform = transforms.Compose([
        transforms.Resize(IMG_SIZE),
        transforms.ToTensor(),
        transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5)),
    ])
    dataset = datasets.CIFAR10(
        root="./data", train=True, download=True, transform=transform)
    return DataLoader(dataset, batch_size=batch_size,
                      shuffle=True, num_workers=2, pin_memory=True,
                      drop_last=True)


def save_checkpoint(model: nn.Module, optimizer: torch.optim.Optimizer,
                    epoch: int, loss: float, path: str) -> None:
    torch.save({
        "epoch":       epoch,
        "model_state": model.state_dict(),
        "optim_state": optimizer.state_dict(),
        "loss":        loss,
    }, path)
    print(f"  Checkpoint saved → {path}")


def load_checkpoint(model: nn.Module, optimizer: torch.optim.Optimizer,
                    path: str) -> Tuple[int, float]:
    ckpt = torch.load(path, map_location=DEVICE)
    model.load_state_dict(ckpt["model_state"])
    optimizer.load_state_dict(ckpt["optim_state"])
    print(f"  Checkpoint loaded ← {path}  (epoch {ckpt['epoch']})")
    return ckpt["epoch"], ckpt["loss"]


def save_image_grid(samples: torch.Tensor, path: str, nrow: int = 4) -> None:
    """Save a grid of images (values in [-1, 1]) to disk."""
    imgs  = (samples.clamp(-1, 1) + 1.0) / 2.0   # → [0,1]
    grid  = make_grid(imgs, nrow=nrow, padding=2)
    npimg = grid.permute(1, 2, 0).cpu().numpy()
    plt.figure(figsize=(8, 8))
    plt.imshow(npimg)
    plt.axis("off")
    plt.tight_layout()
    plt.savefig(path, dpi=120)
    plt.close()
    print(f"  Sample grid saved → {path}")

def train(resume_path: Optional[str] = None) -> None:
    """
    Main training loop.

    - Model: BlockSparseDTM (with dilated chain)
    - Loss:  DSM (denoising score matching) on CIFAR-10 64x64
    - Opt:   AdamW with cosine LR decay
    - Saves checkpoints and sample grids at configurable intervals
    """
    print("\n" + "=" * 60)
    print("  Block-Sparse DTM with Dilated Block Chaining")
    print(f"  Chain dilations  : {CHAIN_DILATIONS}")
    print(f"  Chain hidden dim : {CHAIN_HIDDEN}")
    print(f"  Fine block cond  : 128 (CondProc) + {CHAIN_HIDDEN} (chain) = "
          f"{128 + CHAIN_HIDDEN}")
    print(f"  Device           : {DEVICE}")
    print(f"  Epochs           : {NUM_EPOCHS}")
    print("=" * 60 + "\n")

    loader   = get_cifar10_loader()
    model    = BlockSparseDTM().to(DEVICE)
    optimizer = torch.optim.AdamW(model.parameters(), lr=LR, weight_decay=1e-4)
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(
        optimizer, T_max=NUM_EPOCHS)

    start_epoch = 0
    best_loss   = float("inf")

    if resume_path and os.path.isfile(resume_path):
        start_epoch, best_loss = load_checkpoint(model, optimizer, resume_path)
        start_epoch += 1

    total_params = sum(p.numel() for p in model.parameters())
    print(f"  Total parameters : {total_params:,}")

    for epoch in range(start_epoch, NUM_EPOCHS):
        model.train()
        epoch_loss  = 0.0
        epoch_start = time.time()

        for batch_idx, (imgs, _) in enumerate(loader):
            imgs = imgs.to(DEVICE)

            optimizer.zero_grad()
            # Use the image itself as its own (unconditional) conditioning ref.
            # In a conditional setup this would be a separate guide image.
            loss = model.dsm_loss(imgs, cond_image=imgs)
            loss.backward()
            torch.nn.utils.clip_grad_norm_(model.parameters(), max_norm=1.0)
            optimizer.step()

            epoch_loss += loss.item()

            if batch_idx % 100 == 0:
                print(f"  Epoch {epoch:3d} | Step {batch_idx:4d} "
                      f"| Loss {loss.item():.4f}")

        scheduler.step()
        avg_loss = epoch_loss / len(loader)
        elapsed  = time.time() - epoch_start
        print(f"\nEpoch {epoch:3d} complete — avg loss: {avg_loss:.4f} "
              f"({elapsed:.1f}s)\n")

        # ── Checkpoint ────────────────────────────────────────────────
        if (epoch + 1) % SAVE_INTERVAL == 0 or epoch == NUM_EPOCHS - 1:
            ckpt_path = os.path.join(
                OUTPUT_DIR, f"ckpt_epoch{epoch:04d}.pt")
            save_checkpoint(model, optimizer, epoch, avg_loss, ckpt_path)
            if avg_loss < best_loss:
                best_loss  = avg_loss
                best_path  = os.path.join(OUTPUT_DIR, "ckpt_best.pt")
                save_checkpoint(model, optimizer, epoch, avg_loss, best_path)

        # ── Sample grid ───────────────────────────────────────────────
        if (epoch + 1) % SAMPLE_INTERVAL == 0 or epoch == NUM_EPOCHS - 1:
            model.eval()
            print("  Generating samples...")
            t0      = time.time()
            samples = model.sample(n_samples=16, verbose=False)
            print(f"  Sampling took {time.time() - t0:.1f}s")
            grid_path = os.path.join(
                OUTPUT_DIR, f"samples_epoch{epoch:04d}.png")
            save_image_grid(samples, grid_path, nrow=4)
            model.train()

    print("\nTraining complete.")
    print(f"Outputs saved to: {OUTPUT_DIR}/")

def run_smoke_test() -> None:
    """
    Quick forward-pass smoke test — no data download required.
    Verifies shapes and gradient flow through the full dilated chain path.
    """
    print("\n── Smoke test ──────────────────────────────────────────────")
    model = BlockSparseDTM().to(DEVICE)
    model.train()

    B     = 2
    x0    = torch.randn(B, CHANNELS, IMG_SIZE, IMG_SIZE, device=DEVICE)
    cond  = torch.randn(B, CHANNELS, IMG_SIZE, IMG_SIZE, device=DEVICE)
    t     = torch.randint(0, T_STEPS, (B,), device=DEVICE)

    # Forward: add noise
    x_t, noise = model.add_noise(x0, t)
    print(f"  x_t shape     : {x_t.shape}")
    print(f"  noise shape   : {noise.shape}")

    # Forward: fine block extraction
    fine_blocks = BlockSparseDTM._extract_blocks(
        x_t, FINE_BLOCK_SIZE, FINE_BLOCKS_PER_SIDE)
    print(f"  fine_blocks   : {fine_blocks.shape}")   # [B, 64, 192]

    # Forward: dilated chain
    chain_ctx, tau = model.dilated_chain(fine_blocks)
    print(f"  chain_ctx     : {chain_ctx.shape}")     # [B, 64, 256]
    print(f"  tau           : {tau.shape}")           # [B, 64]

    # Forward: full energy
    x_t_grad = x_t.detach().requires_grad_(True)
    energy   = model.compute_energy(x_t_grad, t, cond)
    print(f"  energy shape  : {energy.shape}")        # [B]
    print(f"  energy values : {energy.detach().cpu().numpy()}")

    # Backward: DSM loss
    loss = model.dsm_loss(x0, cond_image=cond)
    loss.backward()
    print(f"  DSM loss      : {loss.item():.4f}")
    print(f"  Gradients OK  : True")

    # Sampling (1 step only for speed)
    model.eval()
    with torch.no_grad():
        x_t_sample = torch.randn(2, CHANNELS, IMG_SIZE, IMG_SIZE, device=DEVICE)
        t_s        = torch.full((2,), 50, dtype=torch.long, device=DEVICE)
        x_s        = x_t_sample.requires_grad_(True)
        score      = model.compute_score(x_s, t_s)
    print(f"  score shape   : {score.shape}")         # [2, 3, 64, 64]

    print("── Smoke test passed ✓ ─────────────────────────────────────\n")


def run_chain_receptive_field_test() -> None:
    """
    Verify that the dilated chain actually propagates information across
    the full sequence of 64 blocks.  Perturb block 0 and check that the
    change in chain output reaches block 63.
    """
    print("\n── Receptive field test ────────────────────────────────────")
    chain = DilatedBlockChain(
        input_dim=PBITS_PER_FINE_BLOCK,
        hidden_dim=CHAIN_HIDDEN,
        dilations=CHAIN_DILATIONS,
    ).to(DEVICE).eval()

    base   = torch.zeros(1, NUM_FINE_BLOCKS, PBITS_PER_FINE_BLOCK, device=DEVICE)
    perturb = base.clone()
    perturb[0, 0, :] += 1.0   # perturb block 0

    with torch.no_grad():
        h_base, _    = chain(base)
        h_perturb, _ = chain(perturb)

    diff = (h_perturb - h_base).abs()
    last_block_diff = diff[0, -1, :].mean().item()
    print(f"  Mean |Δh| at block 63 due to perturbation at block 0: "
          f"{last_block_diff:.6f}")
    if last_block_diff > 1e-8:
        print("  ✓ Long-range propagation confirmed.")
    else:
        print("  ✗ Warning: perturbation did not reach block 63.")
    print("── Receptive field test complete ───────────────────────────\n")
